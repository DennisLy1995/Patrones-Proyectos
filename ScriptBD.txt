CREATE TABLE USUARIOS(
NOMBRE VARCHAR PRIMARY KEY,
CONTRASENA VARCHAR
);

CREATE TABLE PARTIDAS(
CODIGO_PARTIDA INT IDENTITY(1,1),
PRIMER_JUGADOR VARCHAR FOREIGN KEY REFERENCES USUARIOS(NOMBRE) NOT NULL,
SEGUNDO_JUGADOR VARCHAR FOREIGN KEY REFERENCES USUARIOS(NOMBRE) NOT NULL,
JSON_PARTIDA VARCHAR NOT NULL,
);


SELECT * FROM USUARIOS
SELECT * FROM PARTIDAS
